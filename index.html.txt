<!DOCTYPE html>
<html>
<head>
    <title>My Mailer</title>
</head>
<body>
    <h1>Send an Email</h1>
    <form id="emailForm">
        <input type="email" id="to" placeholder="Recipient Email" required><br>
        <input type="text" id="subject" placeholder="Subject" required><br>
        <textarea id="text" placeholder="Message"></textarea><br>
        <button type="submit">Send</button>
    </form>

    <script>
        document.getElementById('emailForm').addEventListener('submit', async (e) => {
            e.preventDefault();
            let res = await fetch('https://YOUR_RENDER_BACKEND_URL/send', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    to: document.getElementById('to').value,
                    subject: document.getElementById('subject').value,
                    text: document.getElementById('text').value
                })
            });
            let data = await res.json();
            alert(data.success ? 'Email Sent!' : 'Error: ' + data.error);
        });
    </script>
</body>
</html>
